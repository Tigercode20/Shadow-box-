imgElement.onload = function() { processImage(); }; thresholdSlider.addEventListener('input', (e) => { thresholdValue.innerText = e.target.value; if(imgElement.src) processImage(); }); function processImage() { if (!cv.Mat) return; // قراءة الصورة الأصلية let src = cv.imread(imgElement); let gray = new cv.Mat(); let binary = new cv.Mat(); // تحويلها لرمادي ثم أبيض وأسود عتبة (Binary Threshold) cv.cvtColor(src, gray, cv.COLOR_RGBA2GRAY, 0); let threshVal = parseInt(thresholdSlider.value); cv.threshold(gray, binary, threshVal, 255, cv.THRESH_BINARY_INV); let h = binary.rows; let w = binary.cols; // قص الأجزاء الأربعة (أبعاد تقريبية للشكل الهيكلي للهاتف) let rectTop = new cv.Rect(Math.floor(w*0.2), 0, Math.floor(w*0.6), Math.floor(h*0.35)); let rectBottom = new cv.Rect(Math.floor(w*0.2), Math.floor(h*0.65), Math.floor(w*0.6), Math.floor(h*0.35)); let rectLeft = new cv.Rect(0, Math.floor(h*0.2), Math.floor(w*0.35), Math.floor(h*0.6)); let rectRight = new cv.Rect(Math.floor(w*0.65), Math.floor(h*0.2), Math.floor(w*0.35), Math.floor(h*0.6)); let topPart = binary.roi(rectTop); let bottomPart = binary.roi(rectBottom); let leftPart = binary.roi(rectLeft); let rightPart = binary.roi(rectRight); // تدوير الأجزاء المفرودة let topRotated = new cv.Mat(); let leftRotated = new cv.Mat(); let rightRotated = new cv.Mat(); cv.rotate(topPart, topRotated, cv.ROTATE_180); cv.rotate(leftPart, leftRotated, cv.ROTATE_90_COUNTERCLOCKWISE); cv.rotate(rightPart, rightRotated, cv.ROTATE_90_CLOCKWISE); // إنشاء لوحة الخلفية الكبيرة (Canvas) بيضاء اللون let canvasW = Math.floor(w * 1.5); let canvasH = Math.floor(h * 1.5); let canvasMat = new cv.Mat(canvasH, canvasW, cv.CV_8UC1, new cv.Scalar(255)); let centerX = Math.floor(canvasW / 2); let centerY = Math.floor(canvasH / 2); let rectW = Math.floor(w * 0.5); let rectH = Math.floor(h * 0.5); // رسم مربع المركز (أرضية الصندوق) بالرمادي الفاتح let p1 = new cv.Point(centerX - Math.floor(rectW/2), centerY - Math.floor(rectH/2)); let p2 = new cv.Point(centerX + Math.floor(rectW/2), centerY + Math.floor(rectH/2)); cv.rectangle(canvasMat, p1, p2, new cv.Scalar(220), -1); // دمج الأجزاء المدورة في أماكنها حول الصندوق try { // الجزء العلوي let dstTop = canvasMat.roi(new cv.Rect(centerX - Math.floor(topRotated.cols/2), p1.y - topRotated.rows, topRotated.cols, topRotated.rows)); topRotated.copyTo(dstTop); // الجزء السفلي let dstBottom = canvasMat.roi(new cv.Rect(centerX - Math.floor(bottomPart.cols/2), p2.y, bottomPart.cols, bottomPart.rows)); bottomPart.copyTo(dstBottom); // الجزء الأيسر let dstLeft = canvasMat.roi(new cv.Rect(p1.x - leftRotated.cols, centerY - Math.floor(leftRotated.rows/2), leftRotated.cols, leftRotated.rows)); leftRotated.copyTo(dstLeft); // الجزء الأيمن let dstRight = canvasMat.roi(new cv.Rect(p2.x, centerY - Math.floor(rightRotated.rows/2), rightRotated.cols, rightRotated.rows)); rightRotated.copyTo(dstRight); } catch(e) { console.log("خطأ بسيط في الأبعاد أستُثنيت معالجته"); } // عرض النتيجة على الـ Canvas الخاص بـ HTML cv.imshow('canvasOutput', canvasMat); // إظهار زر التحميل downloadBtn.style.display = 'inline-block'; // تنظيف الذاكرة src.delete(); gray.delete(); binary.delete(); topPart.delete(); bottomPart.delete(); leftPart.delete(); rightPart.delete(); topRotated.delete(); leftRotated.delete(); rightRotated.delete(); canvasMat.delete(); } // تفعيل خاصية تحميل الصورة الناتجة عند الضغط على الزر downloadBtn.addEventListener('click', () => { let canvas = document.getElementById('canvasOutput'); let imageURL = canvas.toDataURL("image/png"); let link = document.createElement('a'); link.download = 'shadow_box_layout.png'; link.href = imageURL; link.click(); });
